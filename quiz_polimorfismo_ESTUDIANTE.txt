Q13. (3.5 pts) ¿Cuál de los siguientes es un ejemplo de
     UPCASTING válido e implícito en C++?

  class Animal {
  public:
      virtual void hablar() {}
      virtual ~Animal() {}
  };
  class Perro : public Animal {
  public:
      void ladrar() {}
  };

  Perro p;

  A) Perro* ptr = (Perro*) new Animal();
  B) Animal* a = &p;
  C) Perro* q = dynamic_cast<Perro*>(&p);
  D) Animal a = p;

  Respuesta: B

  ¿Por qué el upcasting es siempre seguro, a diferencia
  del downcasting?
  Porque el upcasting trabaja "de abajo para arriba", es decir, trabaja desde las clases hijas hacia las clase base, 
  la existencia de una clase hija implica la existencia de una clase base.
  Por ejemplo, todo perro es animal, pero no todo animal es perro.

Q14. (3.5 pts) Dado este código, ¿qué imprime el programa?

  #include <iostream>
  using namespace std;

  class Empleado {
  protected:
      string nombre;
  public:
      Empleado(string n) : nombre(n) {}
      virtual void trabajar() {
          cout << nombre << ": trabajo generico" << endl;
      }
      virtual ~Empleado() {}
  };

  class Programador : public Empleado {
  public:
      Programador(string n) : Empleado(n) {}
      void trabajar() override {
          cout << nombre << ": escribe codigo" << endl;
      }
      void hackear() {
          cout << nombre << ": modo hacker!" << endl;
      }
  };

  class Disenador : public Empleado {
  public:
      Disenador(string n) : Empleado(n) {}
      void trabajar() override {
          cout << nombre << ": crea mockups" << endl;
      }
  };

  int main() {
      Empleado* equipo[2];
      equipo[0] = new Programador("Ana");   // upcasting implícito
      equipo[1] = new Disenador("Luis");    // upcasting implícito

      for (int i = 0; i < 2; i++)
          equipo[i]->trabajar();            // polimorfismo

      // Downcasting:
      Programador* p = dynamic_cast<Programador*>(equipo[0]);
      Programador* q = dynamic_cast<Programador*>(equipo[1]);

      if (p) p->hackear();
      if (q) q->hackear();

      delete equipo[0]; delete equipo[1];
      return 0;
  }

  A) Ana: trabajo generico / Luis: trabajo generico /
     Ana: modo hacker! / Luis: modo hacker!

  B) Ana: escribe codigo / Luis: crea mockups /
     Ana: modo hacker!

  C) Ana: escribe codigo / Luis: crea mockups /
     Ana: modo hacker! / Luis: modo hacker!

  D) Error de compilación — equipo[0] es Empleado*,
     no puede llamar hackear()

  Respuesta: B
