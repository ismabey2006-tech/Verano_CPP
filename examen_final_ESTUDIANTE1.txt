      double lectura;
      string unidad;

          cout << "Sensor(" << lectura << ")" << endl;

          cout << "~Sensor(" << lectura << ")" << endl;
      }
      void setLectura(double l) {
          if (l >= 0) lectura = l;
      }
      double getLectura() const { return lectura; }
      void imprimir() const {
          cout << lectura << " " << unidad << endl;
      }
  };

      s.setLectura(0.0);
      s.imprimir();
  }

      Sensor s1(36.6, "C");

      s1.imprimir();
      {
          Sensor s3(0.5, "V");

      return 0;
  }

CONTEXTO:

Estás construyendo el sistema de una biblioteca digital.
La biblioteca tiene distintos tipos de publicaciones.
Todas comparten información básica, pero cada tipo
tiene comportamiento específico al mostrarse y calcular
su duración de préstamo.

────────────────────────────────────────────────────────────────
LO QUE DEBES IMPLEMENTAR:

  1. Clase base abstracta Publicacion:
       Atributos (protected):
         string titulo
         string autor
         int    anio
       Constructor: Publicacion(string titulo,
                                string autor, int anio)
       virtual int diasPrestamo() const = 0
       virtual void imprimir() const
         → imprime: [anio] titulo — autor
                    Prestamo: diasPrestamo() dias
       string getTitulo() const → retorna titulo
       virtual ~Publicacion() {}

  Nota: para imprimir con cout << obj, define FUERA de la clase:
       ostream& operator<<(ostream& os, const Publicacion& p)
         → llama p.imprimir() y retorna os

  2. Clase Libro (hija de Publicacion):
       Atributo (private): int paginas
       Constructor: Libro(string titulo, string autor,
                          int anio, int paginas)
       int diasPrestamo() const override
         → retorna 14 si paginas <= 300, si no retorna 21
       void imprimir() const override
         → llama Publicacion::imprimir()
         → imprime: Paginas: paginas

  3. Clase Revista (hija de Publicacion):
       Atributo (private): int numero
       Constructor: Revista(string titulo, string autor,
                            int anio, int numero)
       int diasPrestamo() const override → retorna 7
       void imprimir() const override
         → llama Publicacion::imprimir()
         → imprime: Numero: numero

  4. Main:
       - Crea un arreglo Publicacion* de 4 elementos
       - Asigna con new: 2 Libros y 2 Revistas
       - Recorre con un bucle e imprime cada publicación
         usando operator<<
       - Escribe en "catalogo.txt" el titulo y los dias
         de préstamo de cada publicación (una línea c/u)
       - Libera la memoria con delete + nullptr
